<template>
  <div>
    <h1>Phones List</h1>

    <div class="cards">
      <div class="card" v-for="item in phones" :key="item._id">
        <h2>{{ item.title }}</h2>
        <p>{{ item.description }}</p>

        <p>
          <strong>Price:</strong> ${{ item.price }}
        </p>

        <p v-if="item.previousPrice">
          <del>Old Price: ${{ item.previousPrice }}</del>
        </p>

        <p><strong>Category:</strong> {{ item.category }}</p>
      </div>
    </div>
  </div>
</template>

<script>
import axios from "axios";

export default {
  name: "PhoneCards",

  data() {
    return {
      phones: [], // API data ayega idhar
    };
  },

  created() {
    this.fetchPhones();
  },

  methods: {
    async fetchPhones() {
      try {
        const response = await axios.get("https://jsonserver.reactbd.com/phone");
        this.phones = response.data;
      } catch (error) {
        console.log("API Error:", error);
      }
    },
  },
};
</script>

  display: grid;
  grid-template-columns: repeat(3, 1fr);
